class BulkMolecules(object):
    """BulkMolecules"""

    def __init__(self, raw_data, sim_data):
        bulkData = np.zeros(
            0,
            dtype=[
                ("id", "U50"),
                ("mass", "{}f8".format(len(sim_data.submass_name_to_index))),
            ],
        )

        # Add units to values
        field_units = {
            "id": None,
            "mass": units.g / units.mol,
        }

        self.bulk_data = UnitStructArray(bulkData, field_units)
